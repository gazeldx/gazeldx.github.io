    $ psql -l # List all databases
    
    # \dt # List all tables.
    
    # \d+ schema_migrations # Show DDL of a table
    
    $ CREATE USER postgres SUPERUSER;# if you got error: ActiveRecord::NoDatabaseError: FATAL:  role "postgres" does not exist
